class CopyFile{
public:
    void copyFiles(){
        fstream myFile;
        fstream archivores;
        myFile.open("archivo.txt",ios::in);
        archivores.open("archivo_resultado.txt",ios::out);
        if(myFile.is_open())
        {
            string line;
            while (getline(myFile,line,','))
            {
                archivores<<line;
            }
        }
        myFile.close();
        archivores.close();
    }
};
